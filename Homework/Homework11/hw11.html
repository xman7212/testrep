def trapezoidal(f, a, b, n):
    h = float(b-a)/n
    result = 0.5*f(a) + 0.5*f(b)
    for i in range(1, n):
        result += f(a + i*h)
    result *= h
    return result

def simps(f, a, b, n):
  sum = 0
  w = (b - a) / n
  for i in range(n + 1):
    x = a + (i * w)
    if x==0:
       continue
    summand = f(x)
    if (i != 0) and (i != n):
      summand *= (2 + (2 * (i % 2)))
    sum += summand
  return ((b - a) / (3 * n)) * sum
